60:     
61:     # If no filter is defined - set all languages selected
62:     session[:language_filter] ||= []

64:       session[:language_filter] = I18n.available_locales.map { |x| x.to_s }
65:     end
66:     
67:     rc = I18n.available_locales.map { |l| 
68:       unless session[:language_filter].include?(l.to_s)
69:         link_to( t("locales.#{l.to_s}".to_sym), 
70:                      set_language_filter_path(l.to_s), :remote => true,
71:                      :title => t(:help_language_filter)
72:                )+sc(:nbsp,:failed).html_safe
73:       else
74:         link_to( t("locales.#{l.to_s}".to_sym),
